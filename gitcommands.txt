######################### BASIC COMMANDS #########################
git add urfile.rmd		# adds urfile.rmd to repo
git add -A 			# adds all all files not in repo
git commit -m "any message"	# "saves" file version
git push origin main		# pushes versions of files to the cloud
git pull origin main		# gets versions of files from the cloud

######################### SETTING UP COMMANDS #########################
# ssh
git init
# must have ssh keys created and linked
git remote add origin git@github.com:luciacarrera/fireProject.git
git pull origin main

# https
git init
git remote add origin https://github.com/luciacarrera/fireProject.git
git pull origin main
